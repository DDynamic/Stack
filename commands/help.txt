msg.delete();

var prefix = (process.env.PREFIX || 's!');

var help = '';

database.query('SELECT * FROM commands', (err, res) => {
    var rows = res.rows;

    for (row in rows) {
        row = rows[row];
        help += `\`` + prefix + ' ' + row['aliases'].split(', ')[0] + `\` ` + row['help'] + `\n`;
    }

    msg.author.send(`**Stack Bot** | info: https://github.com/DDynamic/Stack \n\n` + help);
});
